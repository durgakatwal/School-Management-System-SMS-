## 4. Alternative: More Efficient Backend Approach

If the above is too slow (making many API calls), create a single efficient endpoint:

**In `handlers/sclass.js` - Add this efficient route:**
```jsx
// Get classes with counts in single query
router.get("/with-counts", async (req, res) => {
  try {
    const classes = await Sclass.find({});
    
    const classesWithCounts = await Promise.all(
      classes.map(async (cls) => {
        const studentsCount = await Student.countDocuments({ sclassName: cls._id });
        const subjectsCount = await Subject.countDocuments({ sclassName: cls._id });
        
        return {
          _id: cls._id,
          sclassName: cls.sclassName,
          school: cls.school,
          studentsCount,
          subjectsCount,
          createdAt: cls.createdAt,
          updatedAt: cls.updatedAt
        };
      })
    );
    
    res.json(classesWithCounts);
  } catch (error) {
    console.error("Error fetching classes with counts:", error);
    res.status(500).json({ error: "Failed to fetch classes with counts" });
  }
});
```
